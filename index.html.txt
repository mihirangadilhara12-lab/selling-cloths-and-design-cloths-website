            renderShop() {
                const categories = ['all', 'men', 'women', 'kids', 'accessories'];
                const priceRanges = [
                    { value: 'all', label: 'All Prices' },
                    { value: '0-50', label: 'Under $50' },
                    { value: '50-150', label: '$50 - $150' },
                    { value: '150-300', label: '$150 - $300' },
                    { value: '300+', label: '$300+' }
                ];
                
                let filtered = store.products;
                if (store.currentFilter.category !== 'all') {
                    filtered = filtered.filter(p => p.category === store.currentFilter.category);
                }
                if (store.currentFilter.priceRange !== 'all') {
                    const range = store.currentFilter.priceRange;
                    if (range === '0-50') filtered = filtered.filter(p => p.price < 50);
                    else if (range === '50-150') filtered = filtered.filter(p => p.price >= 50 && p.price < 150);
                    else if (range === '150-300') filtered = filtered.filter(p => p.price >= 150 && p.price < 300);
                    else if (range === '300+') filtered = filtered.filter(p => p.price >= 300);
                }

        // Cart Management
        const cart = {
            toggle() {
                const sidebar = document.getElementById('cart-sidebar');
                const overlay = document.getElementById('cart-overlay');
                sidebar.classList.toggle('open');
                overlay.classList.toggle('hidden');
                document.body.style.overflow = sidebar.classList.contains('open') ? 'hidden' : '';
                this.render();
            },
            
            addItem(productId, qty = 1, size = null, color = null) {
                const product = store.products.find(p => p.id === productId);
                if (!product) return;
                
                const existing = store.cart.find(item => 
                    item.id === productId && 
                    item.size === (size || product.sizes[0]) && 
                    item.color === (color || product.colors[0])
                );
                
                if (existing) {
                    existing.qty += qty;
                } else {
                    store.cart.push({
                        ...product,
                        qty: qty,
                        size: size || product.sizes[0],
                        color: color || product.colors[0],
                        cartId: Date.now()
                    });
                }
                
                this.updateCount();
                this.render();
                showToast('Added to cart successfully');
            },
            
            removeItem(cartId) {
                store.cart = store.cart.filter(item => item.cartId !== cartId);
                this.updateCount();
                this.render();
            },
            
            updateQty(cartId, delta) {
                const item = store.cart.find(i => i.cartId === cartId);
                if (item) {
                    item.qty += delta;
                    if (item.qty <= 0) this.removeItem(cartId);
                    else this.render();
                }
                this.updateCount();
            },
            
            updateCount() {
                const count = store.cart.reduce((sum, item) => sum + item.qty, 0);
                document.getElementById('cart-count').textContent = count;
            },
            
            render() {
                const container = document.getElementById('cart-items');
                const totalEl = document.getElementById('cart-total');
                
                if (store.cart.length === 0) {
                    container.innerHTML = `
                        <div class="text-center py-12 text-gray-400">
                            <i class="fas fa-shopping-bag text-6xl mb-4 opacity-30"></i>
                            <p>Your cart is empty</p>
                            <button onclick="cart.toggle(); router.navigate('shop')" class="mt-4 text-yellow-600 hover:underline text-sm">Start Shopping</button>
                        </div>
                    `;
                    totalEl.textContent = '$0.00';
                    return;
                }
                
                container.innerHTML = store.cart.map(item => `
                    <div class="flex gap-4 bg-gray-50 p-4 rounded-lg slide-in">
                        <img src="${item.image}" class="w-20 h-24 object-cover rounded" alt="${item.name}">
                        <div class="flex-1">
                            <h4 class="font-semibold text-sm mb-1">${item.name}</h4>
                            <p class="text-xs text-gray-500 mb-2">${item.size} / ${item.color}</p>
                            <p class="gold-text font-bold">$${item.price}</p>
                            <div class="flex items-center gap-3 mt-2">
                                <button onclick="cart.updateQty(${item.cartId}, -1)" class="w-6 h-6 rounded border flex items-center justify-center hover:bg-gray-200">-</button>
                                <span class="text-sm font-medium">${item.qty}</span>
                                <button onclick="cart.updateQty(${item.cartId}, 1)" class="w-6 h-6 rounded border flex items-center justify-center hover:bg-gray-200">+</button>
                                <button onclick="cart.removeItem(${item.cartId})" class="ml-auto text-red-500 hover:text-red-700 text-xs">
                                    <i class="fas fa-trash"></i>
                                </button>
                            </div>
                        </div>
                    </div>
                `).join('');
                
                const total = store.cart.reduce((sum, item) => sum + (item.price * item.qty), 0);
                totalEl.textContent = '$' + total.toFixed(2);
            },
            
            checkout() {
                if (store.cart.length === 0) {
                    showToast('Your cart is empty');
                    return;
                }
                this.toggle();
                router.navigate('checkout');
            }
        };
